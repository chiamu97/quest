    <script>
        // --- CONFIGURAZIONE ---
        const CODICE_SEGRETO = "1985"; 
        const MINUTI = 15;             
        // ----------------------

        let audioCtx;
        let tempoRimanente = MINUTI * 60;
        let timerInterval;
        let currentInput = "";
        const display = document.getElementById('display');
        const timeDisplay = document.getElementById('time-display');

        // --- SISTEMA AUDIO (Sintetizzatore) ---
        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
            if (audioCtx.state === 'suspended') {
                audioCtx.resume();
            }
        }

        function playTone(freq, type, duration) {
            if (!audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = type;
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + duration);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + duration);
        }

        function soundBeep() { playTone(800, 'square', 0.1); }
        function soundError() { 
            playTone(150, 'sawtooth', 0.4); 
            setTimeout(() => playTone(100, 'sawtooth', 0.4), 100);
        }
        function soundSuccess() { 
            playTone(600, 'sine', 0.2);
            setTimeout(() => playTone(800, 'sine', 0.2), 200);
            setTimeout(() => playTone(1200, 'sine', 0.4), 400);
        }
        function soundAlarm() {
            // Sirena ciclica
            setInterval(() => {
                playTone(400, 'sawtooth', 0.5);
                setTimeout(() => playTone(300, 'sawtooth', 0.5), 400);
            }, 800);
        }

        // --- LOGICA GIOCO ---
        function initGame() {
            initAudio();
            // Suono di avvio
            playTone(1000, 'sine', 0.5);
            
            document.getElementById('start-overlay').style.display = 'none';
            document.getElementById('timer-container').style.display = 'block';
            document.getElementById('lock-screen').style.display = 'block';
            
            startTimer();
        }

        function startTimer() {
            timerInterval = setInterval(() => {
                tempoRimanente--;
                let m = Math.floor(tempoRimanente / 60);
                let s = tempoRimanente % 60;
                m = m < 10 ? "0" + m : m;
                s = s < 10 ? "0" + s : s;
                timeDisplay.innerText = `${m}:${s}`;

                if (tempoRimanente < 60) {
                    document.getElementById('timer-container').classList.add('timer-panic');
                }
                if (tempoRimanente <= 0) {
                    clearInterval(timerInterval);
                    triggerGameOver();
                }
            }, 1000);
        }

        function triggerGameOver() {
            soundAlarm();
            document.getElementById('lock-screen').style.display = 'none';
            document.getElementById('timer-container').style.display = 'none';
            document.getElementById('game-over').style.display = 'flex';
        }

        function pressKey(num) {
            soundBeep();
            if (currentInput.length < 6) {
                currentInput += num;
                updateDisplay();
            }
        }

        function clearDisplay() {
            soundBeep();
            currentInput = "";
            updateDisplay();
        }

        function updateDisplay() {
            display.value = currentInput; 
        }

        function checkCode() {
            if (currentInput === CODICE_SEGRETO) {
                clearInterval(timerInterval); 
                soundSuccess();
                document.getElementById('lock-screen').style.display = 'none';
                document.getElementById('timer-container').style.display = 'none';
                document.getElementById('secret-content').style.display = 'block';
            } else {
                soundError();
                alert("PIN ERRATO!");
                clearDisplay();
            }
        }
